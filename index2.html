                // Given Q queries, Cheak if the given number is present in the array or not
    // Value of the all the elements in the array is less than 10 to the power 5.
    
    #include <iostream>
    #include <vector>
    using namespace std;
    
    int main() {
    
        vector v;
        int n;
    
        cout << "Enter the size of vector : ";
        cin >> n;
    
        for(int i = 0; i < n; i++) {
            int element;
            cin >> element;
            v.push_back(element);
        }
    
        const int N = 1e5 +10;
        vector freq(N, 0);
    
        for(int i = 0; i < n; i++) {
            freq[v[i]]++;
        }
    
        cout << "Enter queries : ";
        int q;
        cin >> q;
    
        while(q--) {
            int query;
            cin >> query;
            cout << freq[query] << endl;
        }
    
    
        return 0;
    }
